MAJOR update.

# Converted the `langues` columns in the `prestablog_news` and `prestablog_subblock` tables from serialized data to JSON format.

ALTER TABLE `[pre]prestablog_news`
MODIFY COLUMN `langues` TEXT;

UPDATE `[pre]prestablog_news`
SET `langues` = JSON_OBJECT('langues', `langues`)
WHERE JSON_VALID(`langues`) IS NULL;

ALTER TABLE `[pre]prestablog_subblock`
MODIFY COLUMN `langues` TEXT;

UPDATE `[pre]prestablog_subblock`
SET `langues` = JSON_OBJECT('langues', `langues`)
WHERE JSON_VALID(`langues`) IS NULL;

# Updated the `value` field in the `configuration` table for entries 'prestablog_sbr', 'prestablog_sbl', and 'prestablog_commentfb_modosId' to JSON format.

UPDATE `[pre]configuration`
SET `value` = JSON_OBJECT('value', `value`)
WHERE `name` IN ('prestablog_sbr', 'prestablog_sbl', 'prestablog_commentfb_modosId')
AND JSON_VALID(`value`) IS NULL;
